1//! Module for poker hand evaluation and comparison.
2//!
3//! There is a bit of Poker lingo that this file uses. If you find yourself confused by anything,
4//! please reach out to us and ask! The term that will probably be most confusing is a "kicker".
5//! A "kicker", also known as a "side card", is simply a card that does not take part in determine
6//! the rank of a hand. For example, if we want to compare 2 four-of-a-kinds, the fifth card in each
7//! hand would be considered the kicker. Note that kickers are still important in breaking ties.
8//!
9//! A good way to approach this assignment is to first read the code in `src/card.rs`, and then
10//! read all of the starter code in this file (`src/hand.rs`). If you are unfamiliar with Poker, you
11//! may also want to read up on the hands you can make on [Wikipedia]. Finally (as a good rule of
12//! thumb for any programming course), make sure to go over the test cases to make sure you have a
13//! correct idea of what you should implement.
14//!
15//! You are also allowed to change _any_ of the private fields of structs in this file, as long as
16//! you are able to implement the all of the [`Hand`] methods without issue. You are also allowed to
17//! modify the [`PokerHand::solve`] method, but be aware that the test cases are _solely_ using that
18//! method to check the correctness of your implementation, so make sure not to play around with it
19//! too much.
20//!
21//! [Wikipedia]: https://en.wikipedia.org/wiki/List_of_poker_hands

25/// Represents different poker hand rankings with their respective cards. Each variant contains the
26/// relevant cards that make up the hand.
27///
28/// Note to students: When you auto derive [`PartialOrd`] and [`Ord`], the enum variants are ordered
29/// by declaration order. For example, [`HighCard`] will always come before [`OnePair`].
30///
31/// This type has easily derivable comparison traits, as later [`PokerHand`] variants always beat
32/// earlier ones, and each of the variants are able to be compared with themselves.
33///
34/// See the integration tests in `tests/poker_tests.rs` for examples.
35#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
36pub enum PokerHand {
37    HighCard(HighCard),
38    OnePair(OnePair),
39    TwoPair(TwoPair),
40    ThreeOfAKind(ThreeOfAKind),
41    Straight(Straight),
42    Flush(Flush),
43    FullHouse(FullHouse),
44    FourOfAKind(FourOfAKind),
45    StraightFlush(StraightFlush),
46}

48/// Represents a high card hand, consisting of five unpaired cards.
49#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
50pub struct HighCard {
51    /// The highest card rank in the hand.
52    high_card: Rank,
53    /// The remaining four card ranks in descending order.
54    kickers: [Rank; 4],
55}

58#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
59pub struct OnePair {
60    /// The rank of the matched pair of cards.
61    pair: Rank,
62    /// The remaining three card ranks in descending order.
63    kickers: [Rank; 3],
64}

86/// Represents five consecutive cards of different suits, storing only the highest card rank.
87///
88/// Straights can range from Ace-high straights to wheels. A wheel refers to the lowest possible
89/// 5-high straight (Ace, 2, 3, 4, 5). In this case, even though the Ace is normally the highest
90/// card, it counts as a low card (like a 1), making a wheel.
91#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
92pub struct Straight {
93    /// The highest rank in the straight (highest can be an Ace, lowest is a 5 for a wheel).
94    high_card: Rank,
95}
96
97/// Represents five cards of the same suit. The suit of the cards is ignored since we do not need it
98/// for comparison.
99#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
100pub struct Flush {
101    /// The ranks of the five cards in descending order.
102    ranks: [Rank; 5],
103}

105/// Represents a hand containing three cards of one rank and two cards of another rank.
106#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
107pub struct FullHouse {
108    /// The rank of the three matched cards.
109    triple: Rank,
110    /// The rank of the two matched cards.
111    pair: Rank,
112}

114/// Represents a hand containing four cards of the same rank and one kicker.
115#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
116pub struct FourOfAKind {
117    /// The rank shared by all four cards.
118    quad: Rank,
119    /// The remaining unpaired card rank.
120    kicker: Rank,

123/// Represents five consecutive cards of the same suit. The suit of the cards is ignored since we do
124/// not need it for comparison.
125#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
126pub struct StraightFlush {
127    /// The highest rank in the straight (highest can be an Ace, lowest is a 5 for a wheel).
128    high_card: Rank,
129}

131impl PokerHand {
132    /// Given 5 cards as input, creates a `PokerHand` with the correct ranking.
133    ///
134    /// Note that there is definitely a cleaner way to implement Poker hands, but by breaking it
135    /// down into these specific types we can support partial implementation that don't handle every
136    /// single kind of Poker hand!
137    pub fn solve(hand: Hand) -> Self {
138        // Check for a straight flush first (highest ranking hand).
139        if let Some(straight_flush) = hand.get_straight_flush() {
140            return PokerHand::StraightFlush(straight_flush);
141        }
142
143        // Check for a four of a kind.
144        if let Some(four_of_a_kind) = hand.get_four_of_a_kind() {
145            return PokerHand::FourOfAKind(four_of_a_kind);
146        }
147
148        // Check for a full house.
149        if let Some(full_house) = hand.get_full_house() {
150            return PokerHand::FullHouse(full_house);
151        }
152
153        // Check for a flush.
154        if let Some(flush) = hand.get_flush() {
155            return PokerHand::Flush(flush);
156        }
157
158        // Check for a straight.
159        if let Some(straight) = hand.get_straight() {
160            return PokerHand::Straight(straight);
161        }
162
163        // Check for a three of a kind.
164        if let Some(three_of_a_kind) = hand.get_triple() {
165            return PokerHand::ThreeOfAKind(three_of_a_kind);
166        }
167
168        // Check for a two pair.
169        if let Some(two_pair) = hand.get_two_pair() {
170            return PokerHand::TwoPair(two_pair);
171        }
172
173        // Check for a one pair.
174        if let Some(one_pair) = hand.get_one_pair() {
175            return PokerHand::OnePair(one_pair);
176        }
177
178        // If no other hand is found, it's a high card hand.
179        PokerHand::HighCard(HighCard {
180            high_card: hand.cards[0].rank(),
181            kickers: [
182                hand.cards[1].rank(),
183                hand.cards[2].rank(),
184                hand.cards[3].rank(),
185                hand.cards[4].rank(),
186            ],
187        })
188    }
189}

191/// Represents a standard `Hand` of 5 playing [`Card`]s.
192///
193/// Note that the cards will always be stored in sorted descending order (only by [`Rank`]).
194#[derive(Debug)]
195pub struct Hand {
196    /// The cards in the hand.
197    cards: [Card; 5],
198}

200impl Hand {
201    /// Creates a new `Hand` of 5 [`Card`]s.
202    ///
203    /// Stores the cards in reverse (descending) sorted order.
204    ///
205    /// Returns [`None`] if any cards are duplicates.
206    pub fn new(mut cards: [Card; 5]) -> Option<Self> {
207        // Sort in reverse order.
208        cards.sort_by(|a, b| b.cmp(a));
209
210        // Check for any duplicate cards.
211        for i in 0..4 {
212            // Need to also check if the suits are equal since `Card` equality does not consider
213            // suits when checking comparison.
214            if cards[i] == cards[i + 1] && cards[i].suit() == cards[i + 1].suit() {
215                return None;
216            }
217        }
218
219        Some(Self { cards })
220    }
